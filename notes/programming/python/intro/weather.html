forecast_data = {}

""" 
-   For every city name, make an HTTP request to get the lat-long coordinates
-   Make an additional call with the coords. to get the city's 5-day forecast data and save to a dictionary
"""
for city in CITIES:
    params = {
        "q": city,
        "limit": LIMIT,
        "appid": API_KEY
    }
    resp = requests.get(URL, params=params)
    
    if resp.ok:
        json = resp.json() 
    else:
        resp.raise_for_status()

    if type(json) == list:
        lat = json[0]['lat']
        lon = json[0]['lon']

    params = {
        "lat": lat,
        "lon": lon,
        "appid": API_KEY
    }
    resp = requests.get(FORECAST_URL, params=params)

    if resp.ok:
        json = resp.json()
    else:
        resp.raise_for_status()

    forecast_data[city] = json['list']

refined_data = {}

""" 
-   For every city, save the min, max temperatures for each day starting from "tomorrow"
"""
for city, data in forecast_data.items():
    new_day = False
    day = 0
    min_max_data = defaultdict(list)
    for d in data:
        # "Tomorrow" is marked as the first day that starts at 00:00:00
        if d["dt_txt"].split()[-1] == "00:00:00":
            new_day = True 
            day += 1

        # Exclude the extra day 5 data
        if day == 5:
            break

        # Min-max data from every day after starting point should be included 
        if new_day:
            main = d["main"]
            min_max_data[day].append((main["temp_min"], main["temp_max"]))
    refined_data[city] = min_max_data

final_data = {city: {"min": [], "max": []} for city in refined_data.keys()}

""" 
-   Calculate the min, max, min_avg, and max_avg temperatures for every day across every city
"""
for city, days in refined_data.items():
    for day, temps in days.items():
        # Saved the data as min-max pairs so we are just retrieving the appropriate data points
        min_temps = list(map(operator.itemgetter(0), temps))
        max_temps = list(map(operator.itemgetter(1), temps))
        
        # Calculate statistics
        min_temp = min(min_temps)
        max_temp = max(max_temps)

        # Save statistics to final data
        final_data[city]["min"].append(min_temp)
        final_data[city]["max"].append(max_temp)
        
for city, d in final_data.items():
    final_data[city]["min_avg"] = mean(d["min"])
    final_data[city]["max_avg"] = mean(d["max"])

""" 
-   Write the header and the collected statistics into a new CSV file
-   Statistics are rounded to two decimal places
"""
with open(FILENAME, "w") as f:
    header = "City,Min 1,Max 1,Min 2,Max 2,Min 3,Max 3,Min 4,Max 4,Min Avg,Max Avg\n"
    f.write(header)
    
    entries = []
    for city, d in final_data.items():
        min_max_line = [None] * (len(d["min"]) + len(d["max"]))
        min_max_line[::2] = d["min"]
        min_max_line[1::2] = d["max"]
        entry = f'"{city}",{",".join(format(x, ".2f") for x in min_max_line)},{d["min_avg"]:.2f},{d["max_avg"]:.2f}\n'
        entries.append(entry) 
    
    f.writelines(entries)
